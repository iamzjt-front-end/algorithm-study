function preorderTraversal(root: TreeNode | null): number[] {
  const stack = [];
  const res = [];
  if (root) stack.push(root);
  while (stack.length) {
    const peek = stack.pop();
    res.push(peek.val);
    if (peek.right) stack.push(peek.right);
    if (peek.left) stack.push(peek.left);
  }
  return res;
}
