### 二、解答

```ts
function isValid(s: string): boolean {
  // 1. 新建一个栈。
  // 2. 扫描字符串，遇左括号入栈，遇到和栈顶括号类型匹配的右括号就出栈，类型不匹配直接判定为不合法。
  // 3. 最后栈空了就合法，否则不合法。
  if (s.length % 2 !== 0) return false;
  const stack = [];
  for (let i = 0; i < s.length; i++) {
    const c = s[i];
    if (c === '(' || c === '[' || c === '{') {
      stack.push(c);
    } else {
      const t = stack[stack.length - 1];
      if (
        (t === '(' && c === ')') ||
        (t === '[' && c === ']') ||
        (t === '{' && c === '}')
      ) {
        stack.pop();
      } else {
        return false;
      }
    }
  }
  return !Boolean(stack.length);
}
```

### 三、复杂度

🔸 时间复杂度： O(n)

🔸 空间复杂度： O(n)
